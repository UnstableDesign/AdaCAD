import { Draft } from '../model/draft';
import { Operation } from './operation';


export class And extends Operation{

    //specify allowable inputs
    allowed_types = [Draft];

    //allowable input numbers

    constructor(inputs:Array<Draft>){
        super();
        this.name = "and";
    }


    //may have one or many draft inputs
    inputTypeValid(input: any):boolean{
        const f = this.allowed_types.find(el => (el === input.getClass()));
        if(f === undefined) return false;
        return true;
    }

    //adds input, returns boolean for success
    addInput(input: any):boolean{
        if(!this.inputTypeValid(input)) return false;
        this.inputs.push(input);
    }

    removeInput(input: any){
        const ndx = this.inputs.findIndex(el => el.id === input.id);
        this.inputs.splice(ndx, 1);
    }

    perform():Draft{



        const temp:Draft = new Draft(null);
        return temp;
    }

}
